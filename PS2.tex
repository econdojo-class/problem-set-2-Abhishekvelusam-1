\begin{gather}
    f(x) = 
    \begin{cases}
        \frac{2}{a^2}x,  & \text{if }0\leq x\leq a \\
        0, & \text{otherwise}
    \end{cases}
\end{gather}
where $a>0$. Set a value for $a$, simulate various sample sizes, and compare results to the true distribution.

\begin{gather}
    f(x)=\frac{2}{3}e^{-2x}+2e^{-3x}
\end{gather}
